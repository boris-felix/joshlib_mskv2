 12             "up": {
 13                 "down": [0, 1],

 15                 "right": [1, 0],
 16                 "left": [ - 1, 0]

 20                 "up": [0, 1],
 21                 "right": [1, 0],
 22                 "left": [ - 1, 0]

 26 
 27         /** 
 28             @constructs 
 29             @class A 2D Grid for navigation
 30             @param {Object} options Options hash
 31         */
 32         __constructor: function(options) {
 33             this.options = options;
 34             this.grid = options.grid;
 35             this.currentCoords = false;
 36             this.orientation = options.orientation || "up";

 38 
 39         setGrid: function(grid) {

 41             //this.currentCoords = false;

 44         get: function(coords) {

 46         },
 47 
 48         goTo: function(coords) {
 49             if (!this.currentCoords || coords[0] !== this.currentCoords[0] || coords[1] !== this.currentCoords[1]) {
 50                 this.currentCoords = coords;
 51                 if (this.options.onChange) this.options.onChange(this.currentCoords, this.get(this.currentCoords));
 52             }
 53             if (this.options.onSelect) this.options.onSelect(this.currentCoords, this.get(this.currentCoords));
 54         },
 55 
 56         go: function(move) {
 57             var newx = this.moves[this.orientation][move][0] + this.currentCoords[0];
 58             var newy = this.moves[this.orientation][move][1] + this.currentCoords[1];
 59             if (
 60                 (newy < 0 || newy >= this.grid.length || !this.grid[newy]) ||
 61                 (newx < 0 || newx >= this.grid[newy].length || !this.grid[newy][newx])
 62             ) {
 63 
 64                 if (this.options.onExit) this.options.onExit([ - this.moves[this.orientation][move][0], -this.moves[this.orientation][move][1]]);
 65 
 66             } else {
 67                 this.goTo([newx, newy]);
 68             }
 69         }
