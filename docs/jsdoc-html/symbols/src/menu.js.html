1 (function(J,$) {

  6 	J.Menu = J.Class({
  7 		
  8 		inverses:{
  9 		    "down":"up",
 10 		    "up":"down",
 11 		    "next":"prev",
 12 		    "prev":"next"
 13 		},
 14 		
 15 		__constructor:function(app) 
 16 		{
 17 
 18 		    this.app = app;
 19 		    
 20     		this.registers={};
 21     		
 22     		this.id2index = {};
 23     		
 24     		this.data = {};
 25 		    
 26 			this.beingLoaded = {};
 27 			
 28 			var self=this;
 29 			this.app.subscribe("menuGoTo",function(ev,data) {
 30 			    var register=data[0];
 31 			    var path=data[1];
 32 			    
 33 			    //self.app.publish("control",["enter",menuPath || event.currentTarget.id]); ?
 34 			    if (!path) return;
 35 			    
 36 			    //Go to first child
 37 			    if (self.isDirectory(path)) {
 38 			        var async=true
 39 			        self.resolveMoves(path.substring(0,path.length-1),["down"],function(newPath) {
 40 			            //console.log("FChild callback got "+newPath,newPath);
 41 			            async=false;
 42 			            self.setRegister(register,newPath);
 43 			        });
 44 			        //Set the temporary register
 45 			        if (async) {
 46 			            //console.log("FChild callback was async, setting "+path,path);
 47 			            self.setRegister(register,path);
 48 			        }
 49 			    } else {
 50 			        self.setRegister(register,path);
 51 			    }
 52 			    
 53                 
 54 			    
 55 				
 56 			});
 57 
 58 			this.app.subscribe("menuGo",function(ev,data) {
 59 			    var register=data[0];
 60 			    var path=data[1];
 61 			    
 62 			    //self.app.publish("control",["enter",menuPath || event.currentTarget.id]); ?
 63 			    if (!path) return;
 64 			    
 65 			    //data : [ 0 : nom du registre , 1 : chemin  ]
 66 			    var async=true;
 67 			    self.resolveMoves(self.getRegister(register),path,function(newPath) {
 68 			        async=false;
 69 			        self.app.publish("menuGoTo",[register,newPath],true);
 70 			    });
 71 			    
 72 			    if (async && path=="down") {
 73 			        self.setRegister(register,(self.getRegister(register)+"/").replace(/\/\/$/,"/"));
 74 			    }
 75 			        
 76 			});
 77 			
 78 		},
 79 		
 80 		setRegister:function(register,position) {
 81 		    
 82 		    //try to fix https://github.com/joshfire/france24/issues/#issue/41
 83 		    if (!position) return;
 84 		    
 85 		    this.registers[register]=position;
 86 		    this.app.publish("menuChange",[register,position],true);
 87 		},
 88 		
 89 		getRegister:function(register) {
 90 		    return this.registers[register];
 91 		},
 92 		
 93 		compactMoves:function(moves) {
 94 		    if (moves.length<2) return [].concat(moves);
 95 		    var newMoves = [];
 96 		    
 97 		    for (var i=0;i<moves.length;i++) {
 98 		        if ((i<moves.length-1) && moves[i]==this.inverses[moves[i+1]]) {
 99 		            i++;
100 		        } else {
101 		            newMoves.push(moves[i]);
102 		        }
103 		    }
104 		    //console.log("                                         compat",moves,newMoves);
105 		    return newMoves;
106 		    
107 		},
108 		
109 		resolveMoves:function(path,moves,callback) {
110 		    if (typeof moves=="string") {
111 		        moves = [moves];
112 		    } else {
113 		        moves = [].concat(moves);
114 		    }
115 		    //console.log("resolveMoves ",path,moves[0]);
116 		    
117 		    //Can't resolve moves from a directory. Start from its first child.
118 		    if (this.isDirectory(path)) {
119 		        path = this.getDirName(path);
120 		        moves.unshift("down");
121 		    }
122 		    moves = [].concat(this.compactMoves(moves));
123 		    
124 		    var self = this;
125 		    var next = function() {
126 		        if (moves.length==0) {
127 		            //console.log(" = ",path);
128 		            return callback(path);
129 		        }
130 		        var move = moves.shift();
131 		        
132 		        
133 		        //Moving up is easy
134 		        if (move=="up") {
135 		            var dir = self.getDirName(path);
136 		            
137 		            //Not yet at root level?
138 		            if (dir!=="") {
139 		                path = dir;
140 		            }   
141 		            next();
142 		            
143 		        //Next and previous, also easy
144 		        } else if (move=="next" || move=="prev") {
145 		            var dir = self.getDirName(path)+"/";
146 		            var basename = self.getBaseName(path);
147                     var i=self.id2index[dir][basename];
148                     
149                     i+=(move=="next"?1:-1);
150                     
151                     if (i>=0 && i<self.data[dir].length) {
152                         path = dir+self.data[dir][i].id;
153                     }   
154                     next();
155                     
156                 //Now, down is the hard one.
157 		        } else if (move=="down") {
158 		            
159 		            //If we already have registered childs
160 		            if (self.data[path+"/"]) {
161 		                if (self.data[path+"/"]=="loading") {
162 		                    //TODO fire when ready?
163 		                    
164 		                } else {
165 		                    
166 		                    //Go to the first child
167 		                    if (self.data[path+"/"].length>0) {
168 		                        path = path+"/"+self.data[path+"/"][0].id;
169 		                    }
170 		                    next();
171 		                    
172 		                }
173 		                
174 		            //Asynchronous loading
175 		            //(register=="focus" || register=="prefocus" || register=="preload")
176 		            } else {
177 		                var dir = self.getDirName(path)+"/";
178     		            var basename = self.getBaseName(path);
179                         var i=self.id2index[dir][basename];
180 		                
181 		                if (typeof self.data[dir][i]["getChildren"]=="function") {
182 		                
183     		                self.app.publish("menuDataLoading",[path+"/"],true);
184     		                self.beingLoaded[path+"/"]=true;
185 		                
186 		                    //console.log("BL",self.beingLoaded[path+"/"],path);
187 		                    
188     		                self.data[dir][i]["getChildren"](function(children) {
189 		                    
190     		                    delete self.beingLoaded[path+"/"];
191 		                        
192     		                    if (children.length>0) {
193     		                        self.setData(path+"/",children);
194     		                        path = path+"/"+self.data[path+"/"][0].id;
195     		                    }
196     		                    next();
197 		                    
198 		                    
199     		                },self.data[dir][i]);
200 	                    } else {
201 	                        next();
202 	                    }
203 		            }
204 		            
205 		        }
206 		        
207 		    };
208 		    
209 		    next();
210 		    
211 		},
212 		
213 		isDirectory:function(path) {
214 		    return path.charAt(path.length-1)=="/";
215 		},
216 		
217 		//Returns the final component of a pathname
218 		getBaseName:function(path) {
219 		    var tmp = path.split("/");
220 		    return tmp[tmp.length-1];
221 		},
222         
223         //Returns the directory component of a pathname
224 		getDirName:function(path) {
225 		    return path.replace(/\/[^/]*$/,"");
226 		},
227 		
228 		setData:function(path,data) {
229 		    //console.log("set",path);
230 		    if (this.isDirectory(path)) {
231 		        delete this.beingLoaded[path];
232 		        
233 		        this.data[path] = [];
234 		        this.id2index[path] = {};
235 		        for (var i=0;i<data.length;i++) {
236 		            this.id2index[path][data[i].id] = i;
237 		            this.setData(path+data[i].id,data[i]);
238 	            }
239 		        
240 		    //Leaf
241 	        } else {
242 	            
243                 var dir = this.getDirName(path)+"/";
244                 var basename = this.getBaseName(path);
245                 
246                 var children = data["children"];
247                 delete data["children"];
248 
249                 //console.log(" set leaf",path,dir,basename,this.data,this.id2index);
250                 if (this.data[dir]!==null && this.id2index[dir]!==null) {
251                     var i = this.id2index[dir][basename];
252                     //console.log(" set leaf i=",i);
253                     if (i===null) {
254                         throw "can't setData to a new menu element like that for now";
255                     }
256                     
257                     if (!this.data[dir][i]) {
258                         this.data[dir][i] = data;
259                     } else {
260                         for (var elem in data) {
261                             this.data[dir][i][elem] = data[elem];
262                         }
263                     }
264                     
265                 }
266                 //TODO remove children element in stored data struct
267                 if (children) {
268                     this.setData(path+"/",children);
269                 }
270 	        }
271 	        this.sendMenuDataEvent(path);
272 		},
273 		
274 		sendMenuDataEvent:function(path) {
275 		    this.app.publish("menuData",[path,this.getData(path)]);
276 		},
277 		
278 
279         getData:function(path) 
280         {
281             //Directory
282             if (this.isDirectory(path)) {
283                 
284                 if (this.beingLoaded[path]) {
285                     return "loading";
286                 } else {
287                     return this.data[path];
288                 }
289                 
290             //Leaf
291             } else {
292                 
293                 var dir = this.getDirName(path)+"/";
294                 var basename = this.getBaseName(path);
295                 //console.log("getData",dir,this.data,this.id2index);
296                 if (this.beingLoaded[dir]) {
297                     return "loading";
298                 } else if (this.data[dir] && this.id2index[dir]!==null) {
299                     var i = this.id2index[dir][basename];
300                     if (i!==null) {
301                         return this.data[dir][i];
302                     }
303                 }
304                 return null;
305                 
306             }
307             
308         },
309

316

318                     
319                 }
320                 
321                 
322             });
