 46 	// node-pool
 47 	J.Pool = function(factory) {
 48         var self = {};
 49 
 50         var idleTimeoutMillis = factory.idleTimeoutMillis || 30000;
 51         var reapInterval      = factory.reapIntervalMillis || 1000;
 52 
 53         var availableObjects = [];
 54         var objectTimeout    = {};
 55         var waitingClients   = [];
 56         var obj;
 57         var count = 0;
 58         var removeIdleScheduled = false;
 59 
 60         function log(str) {
 61             if (factory.log) {
 62                 console.log("pool " + factory.name + " - " + str);
 63             }
 64         }
 65 
 66         function removeIdle() {
 67             removeIdleScheduled = false;
 68 
 69             var toKeep = [];
 70             var now = new Date().getTime();
 71             for (var i = 0; i < availableObjects.length; i++) {
 72                 var timeout = objectTimeout[availableObjects[i]];
 73                 if (now < timeout) {
 74                     toKeep.push(availableObjects[i]);
 75                 }
 76                 else {
 77                     log("removeIdle() destroying obj - now:" + now + " timeout:" + timeout);
 78                     self.destroy(availableObjects[i]);
 79                 }
 80             }
 81 
 82             availableObjects = toKeep;
 83 
 84             if (availableObjects.length > 0) {
 85                 log("availableObjects.length=" + availableObjects.length);
 86                 scheduleRemoveIdle();
 87             }
 88             else {
 89                 log("removeIdle() all objects removed");
 90             }
 91         }
 92 
 93         function scheduleRemoveIdle() {
 94             if (!removeIdleScheduled) {
 95                 removeIdleScheduled = true;
 96                 setTimeout(removeIdle, reapInterval);
 97             }
 98         }
 99 
100         function dispense() {
101             log("dispense() clients=" + waitingClients.length + " available=" + availableObjects.length);
102             if (waitingClients.length > 0) {
103                 obj = null;
104                 if (availableObjects.length > 0) {
105                     log("dispense() - reusing obj");
106                     obj = availableObjects.shift();
107                     delete objectTimeout[obj];
108                     waitingClients.shift()(obj);
109                 }
110                 else if (count < factory.max) {
111                     count++;
112                     log("dispense() - creating obj - count="+count);
113                     factory.create(function(obj) {
114                         if (waitingClients.length > 0) {
115                             waitingClients.shift()(obj);
116                         }
117                         else {
118                             self.returnToPool(obj);
119                         }
120                     });
121                 }
122             }
123         }
124 
125         self.borrow = function(callback,priority) {
126 
127             if (!priority) priority=0;
128 
129             // high priority, currently only supporting <0 or >=0
130             if (priority<0) {
131                 waitingClients.unshift(callback);
132             } else {
133                 waitingClients.push(callback);
134             }
135 
136             dispense();
137         };
138 
139         self.destroy = function(obj) {
140             count--;
141             factory.destroy(obj);
142         };
143 
144         self.returnToPool = function(obj) {
145             //log("return to pool");
146             availableObjects.push(obj);
147             objectTimeout[obj] = (new Date().getTime() + idleTimeoutMillis);
148             log("timeout: " + objectTimeout[obj]);
149             dispense();
150             scheduleRemoveIdle();
151         };
152 
153         return self;
154     };
