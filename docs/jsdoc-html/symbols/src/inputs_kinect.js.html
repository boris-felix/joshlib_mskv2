227         getCursorWait: function(context, bars, center, innerRadius, size, color) {
228             var animating = true,
229             currentOffset = 0;
230 
231             function makeRGBA() {
232                 return "rgba(" + [].slice.call(arguments, 0).join(",") + ")";
233             }
234             function drawBlock(ctx, barNo) {
235                 ctx.fillStyle = makeRGBA(color.red, color.green, color.blue, (bars + 1 - barNo) / (bars + 1));
236                 ctx.fillRect( - size.width / 2, 0, size.width, size.height);
237             }
238             function calculateAngle(barNo) {
239                 return 2 * barNo * Math.PI / bars;
240             }
241             function calculatePosition(barNo) {
242                 angle = calculateAngle(barNo);
243                 return {
244                     y: (innerRadius * Math.cos( - angle)),
245                     x: (innerRadius * Math.sin( - angle)),
246                     angle: angle
247                 };
248             }
249 
250             function draw(ctx, offset) {
251                 clearFrame(ctx);
252                 ctx.save();
253                 ctx.translate(center.x, center.y);
254                 for (var i = 0; i < bars; i++) {
255                     var curbar = (offset + i) % bars,
256                     pos = calculatePosition(curbar);
257                     ctx.save();
258                     ctx.translate(pos.x, pos.y);
259                     ctx.rotate(pos.angle);
260                     drawBlock(context, i);
261                     ctx.restore();
262                 }
263                 ctx.restore();
264             }
265             function clearFrame(ctx) {
266                 ctx.clearRect(0, 0, ctx.canvas.clientWidth, ctx.canvas.clientHeight);
267             }
268             function nextAnimation() {
269                 if (!animating) {
270                     return;
271                 };
272                 currentOffset = (currentOffset + 1) % bars;
273                 draw(context, currentOffset);
274                 setTimeout(nextAnimation, 50);
275             }
276             nextAnimation(0);
277             return {
278                 stop: function() {
279                     animating = false;
280                     clearFrame(context);
281                 },
282                 start: function() {
283                     currentOffset = parseInt(bars / 2);
284                     animating = true;
285                     nextAnimation(0);
286                 }
287             };
288         }

516     $.idleTimer = function(newTimeout, elem) {
517 
518         // defaults that are to be stored as instance props on the elem
519         var idle = false,
520         //indicates if the user is idle
521         enabled = true,
522         //indicates if the idle timer is enabled
523         timeout = 30000,
524         //the amount of time (ms) before the user is considered idle
525         events = 'mousemove keydown DOMMouseScroll mousewheel mousedown';
526         // activity is one of these events
527 
528         elem = elem || document;
529 
530 
531         var toggleIdleState = function(myelem) {
532 
533             // curse you, mozilla setTimeout lateness bug!
534             if (typeof myelem == 'number') myelem = undefined;
535 
536             var obj = $.data(myelem || elem, 'idleTimerObj');
537 
538             //toggle the state
539             obj.idle = !obj.idle;
540 
541             // reset timeout counter
542             obj.olddate = +new Date;
543 
544             //fire appropriate event
545             // create a custom event, but first, store the new state on the element
546             // and then append that string to a namespace
547             var event = jQuery.Event($.data(elem, 'idleTimer', obj.idle ? "idle": "active") + '.idleTimer');
548 
549             // we dont want this to bubble
550             event.stopPropagation();
551             $(elem).trigger(event);
552         },
553 
554         stop = function(elem) {
555 
556             var obj = $.data(elem, 'idleTimerObj');
557 
558             //set to disabled
559             obj.enabled = false;
560 
561             //clear any pending timeouts
562             clearTimeout(obj.tId);
563 
564             //detach the event handlers
565             $(elem).unbind('.idleTimer');
566         },
567 
568 
569         handleUserEvent = function() {
570 
571             var obj = $.data(this, 'idleTimerObj');
572 
573             //clear any existing timeout
574             clearTimeout(obj.tId);
575 
576 
577 
578             //if the idle timer is enabled
579             if (obj.enabled) {
580 
581 
582                 //if it's idle, that means the user is no longer idle
583                 if (obj.idle) {
584                     toggleIdleState(this);
585                 }
586 
587                 //set a new timeout
588                 obj.tId = setTimeout(toggleIdleState, obj.timeout);
589 
590             }
591         };
592 
593 
594 
595         var obj = $.data(elem, 'idleTimerObj') || new
596         function() {};
597 
598         obj.olddate = obj.olddate || +new Date;
599 
600         //assign a new timeout if necessary
601         if (typeof newTimeout == "number") {
602             timeout = newTimeout;
603         } else if (newTimeout === 'destroy') {
604             stop(elem);
605             return this;
606         } else if (newTimeout === 'getElapsedTime') {
607             return ( + new Date) - obj.olddate;
608         }
609 
610         //assign appropriate event handlers
611         $(elem).bind($.trim((events + ' ').split(' ').join('.idleTimer ')), handleUserEvent);
612 
613 
614         obj.idle = idle;
615         obj.enabled = enabled;
616         obj.timeout = timeout;
617 
618 
619         //set a timeout to toggle state
620         obj.tId = setTimeout(toggleIdleState, obj.timeout);
621 
622         // assume the user is active for the first x seconds.
623         $.data(elem, 'idleTimer', "active");
624 
625         // store our instance on the object
626         $.data(elem, 'idleTimerObj', obj);
627 
628 
629 
630     };
