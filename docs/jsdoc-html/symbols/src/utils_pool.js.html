1 (function(J) {
  2 	
  3 	/** 
  4 	    @constructor
  5 	    @class A Task pool, based on node-pool
  6 	    @param {Object} options Options hash
  7 	*/
  8 	J.Utils.Pool = function(factory) {
  9         var self = {};
 10 
 11         var idleTimeoutMillis = factory.idleTimeoutMillis || 30000;
 12         var reapInterval      = factory.reapIntervalMillis || 1000;
 13 
 14         var availableObjects = [];
 15         var objectTimeout    = {};
 16         var waitingClients   = [];
 17         var obj;
 18         var count = 0;
 19         var removeIdleScheduled = false;
 20 
 21         function log(str) {
 22             if (factory.log) {
 23                 console.log("pool " + factory.name + " - " + str);
 24             }
 25         }
 26 
 27         function removeIdle() {
 28             removeIdleScheduled = false;
 29 
 30             var toKeep = [];
 31             var now = new Date().getTime();
 32             for (var i = 0; i < availableObjects.length; i++) {
 33                 var timeout = objectTimeout[availableObjects[i]];
 34                 if (now < timeout) {
 35                     toKeep.push(availableObjects[i]);
 36                 }
 37                 else {
 38                     log("removeIdle() destroying obj - now:" + now + " timeout:" + timeout);
 39                     self.destroy(availableObjects[i]);
 40                 }
 41             }
 42 
 43             availableObjects = toKeep;
 44 
 45             if (availableObjects.length > 0) {
 46                 log("availableObjects.length=" + availableObjects.length);
 47                 scheduleRemoveIdle();
 48             }
 49             else {
 50                 log("removeIdle() all objects removed");
 51             }
 52         }
 53 
 54         function scheduleRemoveIdle() {
 55             if (!removeIdleScheduled) {
 56                 removeIdleScheduled = true;
 57                 setTimeout(removeIdle, reapInterval);
 58             }
 59         }
 60 
 61         function dispense() {
 62             log("dispense() clients=" + waitingClients.length + " available=" + availableObjects.length);
 63             if (waitingClients.length > 0) {
 64                 obj = null;
 65                 if (availableObjects.length > 0) {
 66                     log("dispense() - reusing obj");
 67                     obj = availableObjects.shift();
 68                     delete objectTimeout[obj];
 69                     waitingClients.shift()(obj);
 70                 }
 71                 else if (count < factory.max) {
 72                     count++;
 73                     log("dispense() - creating obj - count="+count);
 74                     factory.create(function(obj) {
 75                         if (waitingClients.length > 0) {
 76                             waitingClients.shift()(obj);
 77                         }
 78                         else {
 79                             self.returnToPool(obj);
 80                         }
 81                     });
 82                 }
 83             }
 84         }
 85 
 86         self.borrow = function(callback,priority) {
 87 
 88             if (!priority) priority=0;
 89 
 90             // high priority, currently only supporting <0 or >=0
 91             if (priority<0) {
 92                 waitingClients.unshift(callback);
 93             } else {
 94                 waitingClients.push(callback);
 95             }
 96 
 97             dispense();
 98         };
 99 
100         self.destroy = function(obj) {
101             count--;
102             factory.destroy(obj);
103         };
104 
105         self.returnToPool = function(obj) {
106             //log("return to pool");
107             availableObjects.push(obj);
108             objectTimeout[obj] = (new Date().getTime() + idleTimeoutMillis);
109             log("timeout: " + objectTimeout[obj]);
110             dispense();
111             scheduleRemoveIdle();
112         };
113 
114         return self;
115     };
