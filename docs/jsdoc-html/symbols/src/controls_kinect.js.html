196             var animating = true,
197                 currentOffset = 0;
198 
199             function makeRGBA(){
200                 return "rgba(" + [].slice.call(arguments, 0).join(",") + ")";
201             }
202             function drawBlock(ctx, barNo){
203                 ctx.fillStyle = makeRGBA(color.red, color.green, color.blue, (bars+1-barNo)/(bars+1));
204                 ctx.fillRect(-size.width/2, 0, size.width, size.height);
205             }
206             function calculateAngle(barNo){
207                 return 2 * barNo * Math.PI / bars;
208             }
209             function calculatePosition(barNo){
210                 angle = calculateAngle(barNo);
211                 return {
212                     y: (innerRadius * Math.cos(-angle)),
213                     x: (innerRadius * Math.sin(-angle)),
214                     angle: angle
215                 };
216             }
217             
218             function draw(ctx, offset) {
219                 clearFrame(ctx);
220                 ctx.save();
221                 ctx.translate(center.x, center.y);
222                 for(var i = 0; i<bars; i++){
223                     var curbar = (offset+i) % bars,
224                         pos = calculatePosition(curbar);
225                     ctx.save();
226                     ctx.translate(pos.x, pos.y);
227                     ctx.rotate(pos.angle);
228                     drawBlock(context, i);
229                     ctx.restore();
230                 }
231                 ctx.restore();
232             }
233             function clearFrame(ctx) {
234                 ctx.clearRect(0, 0, ctx.canvas.clientWidth, ctx.canvas.clientHeight);
235             }
236             function nextAnimation(){
237                 if (!animating) {
238                     return;
239                 };
240                 currentOffset = (currentOffset + 1) % bars;
241                 draw(context, currentOffset);
242                 setTimeout(nextAnimation, 50);
243             }
244             nextAnimation(0);
245             return {
246                 stop: function (){
247                     animating = false;
248                     clearFrame(context);
249                 },
250                 start: function (){
251                     currentOffset = parseInt(bars/2);
252                     animating = true;

546 $.idleTimer = function(newTimeout, elem){
547   
548     // defaults that are to be stored as instance props on the elem
549     
550     var idle    = false,        //indicates if the user is idle
551         enabled = true,        //indicates if the idle timer is enabled
552         timeout = 30000,        //the amount of time (ms) before the user is considered idle
553         events  = 'mousemove keydown DOMMouseScroll mousewheel mousedown'; // activity is one of these events
554         
555     
556     elem = elem || document;
557     
558     
559         
560     /* (intentionally not documented)
561      * Toggles the idle state and fires an appropriate event.
562      * @return {void}
563      */
564     var toggleIdleState = function(myelem){
565     
566         // curse you, mozilla setTimeout lateness bug!
567         if (typeof myelem == 'number') myelem = undefined;
568     
569         var obj = $.data(myelem || elem,'idleTimerObj');
570         
571         //toggle the state
572         obj.idle = !obj.idle;
573         
574         // reset timeout counter
575         obj.olddate = +new Date;
576         
577         //fire appropriate event
578         
579         // create a custom event, but first, store the new state on the element
580         // and then append that string to a namespace
581         var event = jQuery.Event( $.data(elem,'idleTimer', obj.idle ? "idle" : "active" )  + '.idleTimer'   );
582         
583         // we dont want this to bubble
584         event.stopPropagation();
585         $(elem).trigger(event);            
586     },
587 
588     /**
589      * Stops the idle timer. This removes appropriate event handlers
590      * and cancels any pending timeouts.
591      * @return {void}
592      * @method stop
593      * @static
594      */         
595     stop = function(elem){
596     
597         var obj = $.data(elem,'idleTimerObj');
598         
599         //set to disabled
600         obj.enabled = false;
601         
602         //clear any pending timeouts
603         clearTimeout(obj.tId);
604         
605         //detach the event handlers
606         $(elem).unbind('.idleTimer');
607     },
608     
609     
610     /* (intentionally not documented)
611      * Handles a user event indicating that the user isn't idle.
612      * @param {Event} event A DOM2-normalized event object.
613      * @return {void}
614      */
615     handleUserEvent = function(){
616     
617         var obj = $.data(this,'idleTimerObj');
618         
619         //clear any existing timeout
620         clearTimeout(obj.tId);
621         
622         
623         
624         //if the idle timer is enabled
625         if (obj.enabled){
626         
627           
628             //if it's idle, that means the user is no longer idle
629             if (obj.idle){
630                 toggleIdleState(this);           
631             } 
632         
633             //set a new timeout
634             obj.tId = setTimeout(toggleIdleState, obj.timeout);
635             
636         }    
637      };
638     
639       
640     /**
641      * Starts the idle timer. This adds appropriate event handlers
642      * and starts the first timeout.
643      * @param {int} newTimeout (Optional) A new value for the timeout period in ms.
644      * @return {void}
645      * @method $.idleTimer
646      * @static
647      */ 
648     
649     
650     var obj = $.data(elem,'idleTimerObj') || new function(){};
651     
652     obj.olddate = obj.olddate || +new Date;
653     
654     //assign a new timeout if necessary
655     if (typeof newTimeout == "number"){
656         timeout = newTimeout;
657     } else if (newTimeout === 'destroy') {
658         stop(elem);
659         return this;  
660     } else if (newTimeout === 'getElapsedTime'){
661         return (+new Date) - obj.olddate;
662     }
663     
664     //assign appropriate event handlers
665     $(elem).bind($.trim((events+' ').split(' ').join('.idleTimer ')),handleUserEvent);
666     
667     
668     obj.idle    = idle;
669     obj.enabled = enabled;
670     obj.timeout = timeout;
671     
672     
673     //set a timeout to toggle state
674     obj.tId = setTimeout(toggleIdleState, obj.timeout);
675     
676     // assume the user is active for the first x seconds.
677     $.data(elem,'idleTimer',"active");
678     
679     // store our instance on the object
680     $.data(elem,'idleTimerObj',obj);  
681
