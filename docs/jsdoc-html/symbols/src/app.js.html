124         /**
125          * Subscribes the passed function to the passed message. Every returned token is unique and should be stored if you need to unsubscribe

128          * @param {Function} func The function to call when a new message is published
129          * @returns {String} token for unsubscribing

132             // message is not registered yet
133             if ( !this.subscribes.hasOwnProperty( message ) ){
134                 this.subscribes[message] = [];
135             }
136         
137             // forcing token as String, to allow for future expansions without breaking usage
138             // and allow for easy use as key names for the 'this.subscribes' object
139             var token = (++lastUid).toString();
140             this.subscribes[message].push( { token : token, func : func } );
141         
142             // return token for unsubscribing
143             return token;
144         },
