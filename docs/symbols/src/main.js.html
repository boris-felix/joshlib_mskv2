1 (function( window, $, undefined ) {
  2 
  3 
  4     var lastUid = -1;
  5 
  6 
  7     /** @namespace */
  8 	var J = {
  9 	    basePath:"",
 10 	    Class:$.inherit,
 11 	    
 12 	    /**
 13          * @static
 14          */
 15 	    debugEvents:true,
 16 	    
 17 	    // https://github.com/mroderick/PubSubJS
 18 	    subscribes:{},
 19 	    
 20         /**
 21          *  PubSub.publish( message[, data, sync = false] ) -> Boolean
 22          *  - message (String): The message to publish
 23          *  - data: The data to pass to subscribers
 24          *  - sync (Boolean): Forces publication to be syncronous, which is more confusing, but faster
 25          *  Publishes the the message, passing the data to it's subscribers
 26         **/
 27         publish:function( message, data, sync ){
 28             
 29             if (J.debugEvents) {
 30                 console.log("debugEvents",[message,data,sync]);
 31             }
 32             
 33             // if there are no subscribers to this message, just return here
 34             if ( !J.subscribes.hasOwnProperty( message ) ){
 35                 return false;
 36             }
 37         
 38             var publish = function(){
 39                 var subscribers = J.subscribes[message];
 40                 var throwException = function(e){
 41                     return function(){
 42                         throw e;
 43                     };
 44                 }; 
 45                 for ( var i = 0, j = subscribers.length; i < j; i++ ){
 46                     try {
 47                         subscribers[i].func( message, data );
 48                     } catch( e ){
 49                         setTimeout( throwException(e), 0);
 50                     }
 51                 }
 52             };
 53         
 54             if ( sync === true ){
 55                 publish();
 56             } else {
 57                 setTimeout( publish, 0 );
 58             }
 59             return true;
 60         },
 61         
 62         
 63         /**
 64          *  PubSub.subscribe( message, func ) -> String
 65          *  - message (String): The message to subscribe to
 66          *  - func (Function): The function to call when a new message is published
 67          *  Subscribes the passed function to the passed message. Every returned token is unique and should be stored if you need to unsubscribe
 68         **/
 69         subscribe:function( message, func ){
 70             // message is not registered yet
 71             if ( !J.subscribes.hasOwnProperty( message ) ){
 72                 J.subscribes[message] = [];
 73             }
 74         
 75             // forcing token as String, to allow for future expansions without breaking usage
 76             // and allow for easy use as key names for the 'J.subscribes' object
 77             var token = (++lastUid).toString();
 78             J.subscribes[message].push( { token : token, func : func } );
 79         
 80             // return token for unsubscribing
 81             return token;
 82         },
 83         
 84         /**
 85          *  PubSub.unsubscribe( token ) -> String | Boolean
 86          *  - token (String): The token of the function to unsubscribe
 87          *  Unsubscribes a specific subscriber from a specific message using the unique token
 88         **/
 89         unsubscribe:function( token ){
 90             for ( var m in J.subscribes ){
 91                 if ( J.subscribes.hasOwnProperty( m ) ){
 92                     for ( var i = 0, j = J.subscribes[m].length; i < j; i++ ){
 93                         if ( J.subscribes[m][i].token === token ){
 94                             J.subscribes[m].splice( i, 1 );
 95                             return token;
 96                         }
 97                     }
 98                 }
 99             }
100             return false;
101         }

104     };
