 51 $.fn.idleTimer = function(options){
 52   options  = options || {};
 53   var element  = $(this);
 54 
 55   var idle     = false;  
 56   var timeout  = options.timeout  || 4000; // 3 seconds
 57   var interval = options.interval || 1000; // 1 second
 58   var timeFromLastEvent = 0;
 59   
 60   var reset = function(){ 
 61     if (idle) element.trigger("idle", false);
 62     idle = false;
 63     timeFromLastEvent = 0; 
 64   };
 65   var check = function(){
 66     if (timeFromLastEvent >= timeout) {
 67       reset();
 68       idle = true;
 69       element.trigger("idle", true);
 70     } else {
 71       timeFromLastEvent += interval;
 72     }
 73   };
 74   
 75   element.bind(idleEvents, reset);
 76   var loop = setInterval(check, interval);
 77   element.unload(function(){
 78     clearInterval(loop);
 79   });
 80 };

566 FlareVideo.fn.setupTiming = function(){  
567   var timeToGo = $("<div />");
568   var timeLeft = $("<div />");
569   
570   timeToGo.addClass("timeMin");
571   timeLeft.addClass("timeMax");
572   
573   var pad = function(num) {
574     if (num < 10)
575       return "0" + num;
576     return num;
577   }
578   
579   var secondsFormat = function(sec){
580     var result  = [];
581     
582     var minutes = Math.floor( sec / 60 );
583     var hours   = Math.floor( sec / 3600 );
584     var seconds = (sec == 0) ? 0 : (sec % 60)
585     seconds     = Math.round(seconds);
586         
587     if (hours > 0)
588       result.push(pad(hours));
589     
590     result.push(pad(minutes));
591     result.push(pad(seconds));
592     
593     return result.join(":");
594   };
595   
596   this.ontimeupdate($.proxy(function(){    
597     timeToGo.text(secondsFormat(this.video.getCurrentTime()));    
598     timeLeft.text("-" + secondsFormat(this.video.getEndTime() - this.video.getCurrentTime()))
599   }, this));
600   
601   this.videoElement.one("canplay", $.proxy(function(){
602     this.videoElement.trigger("timeupdate");
603   }, this));
604   
605   this.controls.append(timeToGo);
606   this.controls.append(timeLeft);
607 };
